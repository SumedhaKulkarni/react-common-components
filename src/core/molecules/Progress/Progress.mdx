## Animated Progress Circle
<Playground>
  {() => {
    let textValue = <div>Complete!</div>
    let [percent, setPercent] = useState(50)
    return (
      <div>
        <div>
          <Progress percent={percent} type="circle" strokeSize={10} width={200} />
          <Progress percent={percent} messageInside />
        </div>
        <Button handleClick={() => setPercent(percent+1)}>Increment</Button>
        Percent: {percent}
      </div>
    )
  }}
</Playground>
